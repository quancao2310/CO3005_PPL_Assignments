func swap(number arr[10], number i, number j) begin
    if ((i < 0) or (i >= 10) or (j < 0) or (j >= 10) or (i = j)) return
    var tmp <- arr[i]
    arr[i] <- arr[j]
    arr[j] <- tmp
end

func selection_sort(number arr[10]) begin
    number i <- 0
    for i until i = 9 by 1 begin
        number min_idx <- i
        number j <- i + 1
        for j until j = 10 by 1
            if (arr[j] < arr[min_idx]) min_idx <- j
        swap(arr, i, min_idx)
    end
end

func main() begin
    dynamic arr <- [5.71, 1.69, 3.49, 2.65, 4.28, 7.84, 1.49, 5.52, 2.48, 7.87]
    selection_sort(arr)
    writeNumber(arr[0])
    var i <- 1
    for i until i = 10 by 1 begin
        writeString(" ")
        writeNumber(arr[i])
    end
end
