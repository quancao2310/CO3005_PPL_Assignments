func init()
func a()
func b()
func c()
dynamic i
dynamic j
dynamic k

func main() begin
    init()
    number x1 <- a()[i]
    bool x2 <- b()[i, k]
    string x3 <- c()[j, i]
    
    writeNumber(x1)
    writeBool(x2)
    writeString(x3)
    writeNumber(a()[k])
end

func a() return [5, 4, 3, 2, 1]
func b()
    return [[a()[0] > a()[1], c()[0, 0] == "PPL"], [false, true], [2 = 2, 1 != 3]]
func c() return [["PPL", "is"], ["very", "hard"]]
func init() begin
    i <- 0
    j <- 1
    k <- 1
end
